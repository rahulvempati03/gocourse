	// var age int
	// var name string = "john"

	// count := 10      // short declaration, only inside functions, go for notation

	// middleName = "smith" // re-assigning package level variable // this works with "="
	middleName := "smith" // re-declaring and assigning a new value using short declaration operator ":="
	fmt.Println(middleName)
	// Default values
	// Numeric types: 0
	// String: ""
	// Boolean: false
	// Pointers,slices, maps, interfaces nd structs : nil


	// -- SCOPES 
	// go has block scope, variables declared inside a block are not accessible outside the block
	{
		var blockVar = "I am inside a block"
		fmt.Println(blockVar)
	}
	// fmt.Println(blockVar) // this will cause an error, as blockVar is not accessible here
}
